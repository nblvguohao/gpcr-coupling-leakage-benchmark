1. **交叉注意力机制实现**（2天）
   ```python
   # 需要实现
   class CrossAttention(nn.Module):
       def __init__(self, dim=320):
           super().__init__()
           self.W_Q = nn.Linear(dim, dim)
           self.W_K = nn.Linear(dim, dim)
           self.W_V = nn.Linear(dim, dim)
           self.scale = torch.sqrt(torch.FloatTensor([dim]))
       
       def forward(self, opsin_emb, gnaq_emb):
           # opsin_emb: [batch, 320]
           # gnaq_emb: [batch, 320]
           Q = self.W_Q(opsin_emb)  # [batch, 320]
           K = self.W_K(gnaq_emb)   # [batch, 320]
           V = self.W_V(gnaq_emb)   # [batch, 320]
           
           attention = torch.softmax(
               torch.matmul(Q, K.transpose(-2, -1)) / self.scale, 
               dim=-1
           )  # [batch, batch]
           
           out = torch.matmul(attention, V)  # [batch, 320]
           return out, attention
   ```

2. **对比学习损失实现**（2天）
   ```python
   # 实现InfoNCE loss
   class ContrastiveLoss(nn.Module):
       def __init__(self, temperature=0.5):
           super().__init__()
           self.temperature = temperature
       
       def forward(self, z_i, z_j, labels):
           # z_i, z_j: 同功能不同物种的样本对
           # labels: 1表示正样本对，0表示负样本对
           similarity = F.cosine_similarity(z_i, z_j, dim=-1)
           loss = -torch.mean(
               labels * torch.log(torch.exp(similarity / self.temperature) / 
               torch.sum(torch.exp(similarity / self.temperature)))
           )
           return loss
   ```
